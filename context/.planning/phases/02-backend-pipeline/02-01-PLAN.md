```typescript
// FetchResult type
export interface FetchResult {
  ok: true;
  html: string;
  finalUrl: string;  // After redirects
  contentType: string;
} | {
  ok: false;
  error: 'FETCH_FAILED' | 'TIMEOUT' | 'INVALID_URL' | 'NOT_HTML';
  message: string;
}

// fetchArticle function
export async function fetchArticle(url: string, timeoutMs: number = 5000): Promise<FetchResult>
```

Implementation requirements:
1. Validate URL format using URL constructor (catch invalid URLs)
2. Use native fetch with:
   - redirect: 'follow' (per ING-03)
   - signal: AbortSignal.timeout(timeoutMs)
   - headers: { 'User-Agent': 'BlindSpot/1.0 (+https://blindspot.app)', 'Accept': 'text/html' }
3. Check response.ok and content-type includes 'text/html'
4. Return { ok: true, html: await response.text(), finalUrl: response.url, contentType }
5. Handle errors:
   - AbortError/TimeoutError -> TIMEOUT
   - TypeError (network) -> FETCH_FAILED
   - Invalid URL -> INVALID_URL
   - Non-HTML content-type -> NOT_HTML

Create backend/src/services/extractor.ts:

```typescript
import { Readability } from '@mozilla/readability';
import { JSDOM } from 'jsdom';

export interface ExtractionResult {
  ok: true;
  article: {
    title: string;
    content: string;
    textContent: string;
    excerpt: string;
    byline: string | null;
    publishedTime: string | null;
    siteName: string | null;
  };
  partial: boolean;  // true if paywall/degraded mode (per EXT-03)
} | {
  ok: false;
  error: 'EXTRACTION_FAILED' | 'NO_CONTENT' | 'PAYWALL';
  message: string;
  partial_content?: { title: string; excerpt: string };  // Degraded mode fallback
}

export function extractArticle(html: string, url: string): ExtractionResult
```

Implementation requirements:
1. Parse HTML with JSDOM: `new JSDOM(html, { url })`
2. Extract with Readability: `new Readability(doc.window.document).parse()`
3. Paywall detection (per 02-CONTEXT.md "Combined" approach):
   - Check for paywall indicators: class/id containing 'paywall', 'subscription', 'premium-content'
   - Check for truncated content: textContent < 500 chars after strip
   - If detected: return degraded mode with title + meta description + first paragraphs
4. If Readability returns null or empty content -> NO_CONTENT error with partial_content fallback
5. Extract metadata:
   - title: article.title or og:title or document.title
   - publishedTime: article:published_time meta or time element
   - siteName: og:site_name meta
6. Set partial: true if content < 1000 chars or paywall indicators present
  </action>
